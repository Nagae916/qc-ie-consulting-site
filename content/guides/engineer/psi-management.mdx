  <script>
    // --- Elements ---
    const sales = document.getElementById('sales');
    const i0 = document.getElementById('initialStock');
    const it = document.getElementById('targetStock');
    const salesValue = document.getElementById('salesValue');
    const i0Value = document.getElementById('initialStockValue');
    const itValue = document.getElementById('targetStockValue');
    const prodOut = document.getElementById('productionResult');
    const prodFormula = document.getElementById('productionFormula');

    // bars
    const barInit = document.getElementById('barInit');
    const barProd = document.getElementById('barProd');
    const barSales = document.getElementById('barSales');
    const barEOI = document.getElementById('barEOI');

    function fmt(n){ return Math.max(0, n|0); }

    function updatePSI(){
      const s = parseInt(sales.value,10);
      const iInit = parseInt(i0.value,10);
      const iTarget = parseInt(it.value,10);
      const p = Math.max(0, s + iTarget - iInit);

      salesValue.textContent = s;
      i0Value.textContent = iInit;
      itValue.textContent = iTarget;

      prodOut.textContent = p;
      prodFormula.textContent = `生産量 = ${s} + ${iTarget} − ${iInit}`;

      // 比率を計算（行ごとに最大の合計でスケーリング）
      const supply = iInit + p;
      const demand = s + iTarget;
      const maxRow = Math.max(supply, demand, 1);

      const w = (v)=> (v/maxRow*100).toFixed(1) + '%';

      barInit.style.width = w(iInit);
      barProd.style.width = w(p);
      barSales.style.width = w(s);
      barEOI.style.width = w(iTarget);
    }

    [sales, i0, it].forEach(el => el.addEventListener('input', updatePSI));
    updatePSI();

    // --- Vocabulary ---
    const vocabulary = [
      { term:'PSI（生産・販売・在庫）', def:'Production / Sales / Inventory の頭文字。3指標を統合的に管理する考え方。' },
      { term:'適正在庫', def:'欠品を起こさず、かつ過剰在庫によるコスト増も招かない在庫水準。' },
      { term:'安全在庫', def:'需要変動や供給遅延に備える緩衝在庫。' },
      { term:'期首在庫 / 期末在庫', def:'期間の始まり/終わりの在庫量。期末在庫は次期の期首在庫となる。' },
    ];
    const acc = document.getElementById('accordion');
    vocabulary.forEach(v=>{
      const wrap = document.createElement('div');
      wrap.className = 'border border-slate-200 rounded-lg';
      wrap.innerHTML = `
        <button class="accordion-header w-full text-left p-4 font-semibold bg-slate-50 hover:bg-slate-100 flex justify-between items-center">
          <span>${v.term}</span><span class="transform transition-transform duration-300">▼</span>
        </button>
        <div class="accordion-content"><p class="p-4 pt-0 text-slate-600">${v.def}</p></div>`;
      acc.appendChild(wrap);
    });
    document.querySelectorAll('.accordion-header').forEach(btn=>{
      btn.addEventListener('click', ()=>{
        const c = btn.nextElementSibling;
        const icon = btn.querySelector('span:last-child');
        if(c.style.maxHeight){ c.style.maxHeight=null; icon.style.transform='rotate(0deg)'; }
        else{ document.querySelectorAll('.accordion-content').forEach(x=>x.style.maxHeight=null);
              document.querySelectorAll('.accordion-header span:last-child').forEach(i=>i.style.transform='rotate(0deg)');
              c.style.maxHeight=c.scrollHeight+'px'; icon.style.transform='rotate(180deg)'; }
      });
    });

    // --- Quiz ---
    const quizData = [
      {
        q:"PSI管理の主目的で最も適切なのは？",
        opts:["販売利益の最大化を最優先とする","品質向上のため在庫を常に厚く持つ","P・S・Iを統合管理し過剰在庫と欠品を回避","生産コストの最小化を最優先し欠品は許容"],
        a:2,
        why:"PSIの核心は3要素のバランス最適化により、欠品と過剰在庫を同時に避けることです。"
      },
      {
        q:"期首在庫150、販売800、目標期末在庫200。必要生産量は？",
        opts:["650","750","850","950"],
        a:2,
        why:"生産量 = 販売量 + 期末在庫 − 期首在庫 = 800 + 200 − 150 = 850。"
      },
      {
        q:"安全在庫の主目的として最も適切なのは？",
        opts:["廃棄コスト削減","需要変動や供給遅延による欠品回避","過剰在庫で常時稼働","保管費最小化でCF最大化"],
        a:1,
        why:"不確実性に備え欠品を防ぐためのバッファが安全在庫です。"
      },
      {
        q:"期首在庫が過剰だった。適切な調整は？",
        opts:["当期の生産量を削減し在庫水準を調整","販売目標を上げ生産量も増やす","期首在庫を無視して計画する","販売単価を上げて販売量を落とす"],
        a:0,
        why:"過剰在庫の直接的対策は生産量の抑制（または販売増）です。"
      }
    ];

    const quizWrap = document.getElementById('quiz-container');
    const scoreEl = document.getElementById('score');
    const totalEl = document.getElementById('totalQuestions');
    const resetBtn = document.getElementById('resetQuiz');
    let score = 0;

    function renderQuiz(){
      quizWrap.innerHTML='';
      score = 0; scoreEl.textContent=0; totalEl.textContent = quizData.length;

      quizData.forEach((it, idx)=>{
        const box = document.createElement('div');
        box.className='p-4 border border-slate-200 rounded-lg';
        const opts = it.opts.map((o,i)=>(
          `<label class="block p-2 rounded hover:bg-slate-100 cursor-pointer">
             <input type="radio" name="q${idx}" value="${i}" class="mr-2">${o}
           </label>`
        )).join('');
        box.innerHTML = `
          <p class="font-semibold mb-3">${idx+1}. ${it.q}</p>
          <div class="space-y-1">${opts}</div>
          <button class="check-answer mt-3 bg-slate-200 hover:bg-slate-300 text-slate-700 text-sm font-bold py-1 px-3 rounded" data-i="${idx}">答えを確認</button>
          <div class="feedback mt-2 text-sm p-3 rounded-md hidden"></div>`;
        quizWrap.appendChild(box);
      });

      document.querySelectorAll('.check-answer').forEach(btn=>{
        btn.addEventListener('click', e=>{
          const i = +e.currentTarget.dataset.i;
          const sel = document.querySelector(`input[name="q${i}"]:checked`);
          const fb = e.currentTarget.nextElementSibling;
          if(!sel){
            fb.textContent='選択肢を選んでください。';
            fb.className='feedback mt-2 text-sm p-3 rounded-md bg-yellow-100 text-yellow-800';
            fb.classList.remove('hidden'); return;
          }
          const ok = (+sel.value === quizData[i].a);
          if(ok){
            fb.innerHTML = `<b class="text-green-700">正解！</b> ${quizData[i].why}`;
            fb.className='feedback mt-2 text-sm p-3 rounded-md bg-green-100 text-green-800';
            if(!e.currentTarget.disabled){ score++; scoreEl.textContent=score; }
          }else{
            fb.innerHTML = `<b class="text-red-700">不正解。</b> 正解は「${quizData[i].opts[quizData[i].a]}」。<br>${quizData[i].why}`;
            fb.className='feedback mt-2 text-sm p-3 rounded-md bg-red-100 text-red-800';
          }
          e.currentTarget.disabled = true;
          document.querySelectorAll(`input[name="q${i}"]`).forEach(inp=>inp.disabled=true);
          fb.classList.remove('hidden');
        });
      });
    }

    resetBtn.addEventListener('click', renderQuiz);
    renderQuiz();
  </script>
