    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // PDCA Interaction
            const pdcaItems = document.querySelectorAll('.pdca-item');
            const pdcaDetails = document.querySelectorAll('.pdca-details-card');

            pdcaItems.forEach(item => {
                item.addEventListener('click', () => {
                    pdcaItems.forEach(i => {
                        i.classList.remove('active');
                        i.classList.replace('bg-sky-500', 'bg-slate-200');
                        i.classList.replace('text-white', 'text-slate-800');
                    });
                    item.classList.add('active');
                    item.classList.replace('bg-slate-200', 'bg-sky-500');
                    item.classList.replace('text-slate-800', 'text-white');

                    const targetId = item.dataset.target;
                    pdcaDetails.forEach(card => {
                        if (card.id === targetId) card.classList.add('active');
                        else card.classList.remove('active');
                    });
                });
            });

            // Quiz Logic
            const quizData = {
                q1: { correct: 'A', rationale: '正解です！日常管理は、PDCAサイクルを回すことで、現状の維持とさらなる改善を継続的に行うことを目的としています。' },
                q2: { correct: 'C', rationale: '正解です！日常管理は、TQMの理念である全員参加と継続的改善を、日々の業務を通じて実践する具体的な活動として位置づけられます。' }
            };

            const quizItems = document.querySelectorAll('.quiz-item');
            quizItems.forEach(item => {
                const radios = item.querySelectorAll('input[type="radio"]');
                const quizId = item.querySelector('.space-y-2').dataset.quizId;
                const feedbackEl = document.getElementById(`feedback-${quizId}`);

                radios.forEach(radio => {
                    radio.addEventListener('change', () => {
                        const selectedAns = radio.value;
                        if (selectedAns === quizData[quizId].correct) {
                            feedbackEl.textContent = quizData[quizId].rationale;
                            feedbackEl.className = 'quiz-feedback correct';
                        } else {
                            feedbackEl.textContent = '不正解です。もう一度、各コンセプトの説明を読み返してみましょう。';
                            feedbackEl.className = 'quiz-feedback incorrect';
                        }
                        feedbackEl.style.display = 'block';
                        radios.forEach(r => r.disabled = true);
                    });
                });
            });
        });
    </script>
